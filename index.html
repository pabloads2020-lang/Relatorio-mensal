async function extrairDadosPDF(file) { let ab = await file.arrayBuffer(); let pdf = await pdfjsLib.getDocument({ data: ab }).promise; let textoCompleto = ''; for (let p = 1; p <= pdf.numPages; p++) { let page = await pdf.getPage(p); let textContent = await page.getTextContent(); for (let item of textContent.items) { if (item.str && item.str.trim()) { textoCompleto += item.str + ' '; } } } dadosCarteira = { alocacoes: {}, retornos: {}, rentabilidadeMensal: null, pctCdi: null, dataReferencia: "", patrimonio: 0, mesReferencia: "", anoReferencia: "" }; // ==================== DATA (20/04/2026) ==================== let dataMatch = textoCompleto.match(/Data de Referência[^\d]*(\d{2})\/(\d{2})\/(\d{4})/i); if (!dataMatch) dataMatch = textoCompleto.match(/(\d{2})\/(\d{2})\/(\d{4})/); if (dataMatch) { let meses = ["Janeiro","Fevereiro","Março","Abril","Maio","Junho","Julho","Agosto","Setembro","Outubro","Novembro","Dezembro"]; dadosCarteira.mesReferencia = meses[parseInt(dataMatch[2]) - 1]; dadosCarteira.anoReferencia = dataMatch[3]; dbg(`📅 Data: ${dadosCarteira.mesReferencia}/${dadosCarteira.anoReferencia}`); } // ==================== RENTABILIDADE MÊS ==================== // Procura por "RENTABILIDADE MÊS: 1,40%" let rentMesMatch = textoCompleto.match(/RENTABILIDADE\s+M[ÊE]S:\s*(\d+),(\d{2})%/i); if (rentMesMatch) { dadosCarteira.rentabilidadeMensal = parseFloat(rentMesMatch[1] + '.' + rentMesMatch[2]); dbg(`📈 Rentabilidade Mensal: ${dadosCarteira.rentabilidadeMensal}%`); } else { // Fallback: busca na tabela de referências "Portfólio 1 , 40 %" let refMatch = textoCompleto.match(/Portfólio\s+(\d+)\s*,\s*(\d{2})\s*%/i); if (refMatch) { dadosCarteira.rentabilidadeMensal = parseFloat(refMatch[1] + '.' + refMatch[2]); dbg(`📈 Rentabilidade (fallback): ${dadosCarteira.rentabilidadeMensal}%`); } } // ==================== %CDI ==================== // Procura na tabela RESUMO: "MÊS R$ 28.978,37 1,40% 197,26%" let cdiMatch = textoCompleto.match(/M[ÊE]S[^\d]*R\$\s*[\d\.]+,\d{2}\s+(\d+),(\d{2})%\s+(\d+),(\d{2})%/i); if (cdiMatch) { dadosCarteira.pctCdi = parseFloat(cdiMatch[3] + '.' + cdiMatch[4]); dbg(`📊 %CDI: ${dadosCarteira.pctCdi}%`); } // ==================== PATRIMÔNIO ==================== let patMatch = textoCompleto.match(/PATRIM[OÔ]NIO\s+TOTAL\s+BRUTO:\s*R\$\s*([\d\.]+),(\d{2})/i); if (patMatch) { let valor = patMatch[1].replace(/\./g, ''); dadosCarteira.patrimonio = parseFloat(valor + '.' + patMatch[2]); dbg(`💰 Patrimônio: R$ ${dadosCarteira.patrimonio.toFixed(2)}`); } // ==================== ESTRATÉGIAS ==================== // Procura a seção "Estratégias" e pega o texto let estrategiasIndex = textoCompleto.indexOf('Estratégias'); if (estrategiasIndex !== -1) { let secao = textoCompleto.substring(estrategiasIndex, estrategiasIndex + 2000); dbg('\n=== SEÇÃO Estratégias (bruta) ==='); dbg(secao); // Divide em linhas let linhas = secao.split(/\n/); for (let linha of linhas) { // Padrão: "Pós Fixado ( 36 , 75 %) R $ 761.750,29 0 , 66 %" // Remove espaços extras e procura números let linhaLimpa = linha.replace(/\s+/g, ' ').trim(); // Padrão: nome ( xx,xx% ) ... xx,xx% let regex = /^([A-Za-zÀ-Úà-û\s]+?)\s*\(\s*(\d+)\s*,\s*(\d+)\s*%\).*?(\d+)\s*,\s*(\d+)\s*%/i; let match = linhaLimpa.match(regex); if (match && !match[1].includes('Estratégias') && !match[1].includes('Saldo')) { let nome = match[1].trim(); let alocacao = parseFloat(match[2] + '.' + match[3]); let retorno = parseFloat(match[4] + '.' + match[5]); // Ajusta nome para versão amigável if (nome === 'Pós Fixado') nome = 'Pós-Fixado'; if (nome === 'Pré Fixado') nome = 'Pré-Fixado'; if (nome === 'In fl ação') nome = 'Inflação'; if (nome === 'Renda Variável Brasil') nome = 'Renda Variável Brasil'; if (nome === 'Renda Variável Global') nome = 'Renda Variável Global'; if (nome === 'Renda Fixa Global') nome = 'Renda Fixa Global'; if (alocacao > 0 && alocacao <= 100 && !dadosCarteira.alocacoes[nome]) { dadosCarteira.alocacoes[nome] = alocacao; dadosCarteira.retornos[nome] = retorno; dbg(`✅ ${nome}: ${alocacao}% da carteira, retorno: ${retorno}%`); } } } } // Fallback: busca linha por linha no texto completo let linhas = textoCompleto.split(/\n/); for (let linha of linhas) { if (linha.includes('( ') && linha.includes('%)') && linha.includes('R $')) { let linhaLimpa = linha.replace(/\s+/g, ' ').trim(); let regex = /([A-Za-zÀ-Úà-û\s]+?)\s*\(\s*(\d+)\s*,\s*(\d+)\s*%\).*?(\d+)\s*,\s*(\d+)\s*%/i; let match = linhaLimpa.match(regex); if (match && !match[1].includes('Estratégias') && !dadosCarteira.alocacoes[match[1].trim()]) { let nome = match[1].trim(); let alocacao = parseFloat(match[2] + '.' + match[3]); let retorno = parseFloat(match[4] + '.' + match[5]); if (alocacao > 0 && alocacao <= 100) { dadosCarteira.alocacoes[nome] = alocacao; dadosCarteira.retornos[nome] = retorno; dbg(`✅ (fallback) ${nome}: ${alocacao}% | retorno: ${retorno}%`); } } } } dbg('\n=== RESUMO FINAL ==='); dbg(`Data: ${dadosCarteira.mesReferencia}/${dadosCarteira.anoReferencia}`); dbg(`Rentabilidade: ${dadosCarteira.rentabilidadeMensal !== null ? dadosCarteira.rentabilidadeMensal + '%' : 'não encontrada'}`); dbg(`%CDI: ${dadosCarteira.pctCdi !== null ? dadosCarteira.pctCdi + '%' : 'não encontrado'}`); dbg(`Patrimônio: ${dadosCarteira.patrimonio > 0 ? 'R$ ' + dadosCarteira.patrimonio : 'não encontrado'}`); dbg('Alocações:', dadosCarteira.alocacoes); dbg('Retornos:', dadosCarteira.retornos); return { alocacoes: dadosCarteira.alocacoes, patrimonio: dadosCarteira.patrimonio }; }
